94  void calcGain(const WeightParam & weight_param)
95  {
96  // 0. Calculate discrete system matrices
97  if(model_->dt_ != horizon_dt_)
98  {
99  model_->calcDiscMatrix(horizon_dt_);
100  }
101 
102  const Eigen::Matrix<double, StateDim, StateDim> & A = model_->Ad_;
103  const Eigen::Matrix<double, StateDim, InputDim> & B = model_->Bd_;
104  const Eigen::Matrix<double, OutputDim, StateDim> & C = model_->C_;
105 
106  Eigen::Matrix<double, OutputDim, OutputDim> Q = weight_param.output.asDiagonal();
107  Eigen::Matrix<double, InputDim, InputDim> R = weight_param.input.asDiagonal();
108  Eigen::Matrix<double, InputDim, InputDim> Rinv = weight_param.input.cwiseInverse().asDiagonal();
109 
110  // int stateDim = model_->stateDim();
111  int inputDim = model_->inputDim();
112  int outputDim = model_->outputDim();
113 
114  // 1. Calculate P by solving the Discrete-time Algebraic Riccati Equation
115  // (DARE) ref. https://scicomp.stackexchange.com/a/35394
116  int iterMax = 10000;
117  double relNormThre = 1e-8;
118  Eigen::Matrix<double, StateDim, StateDim> A0 = A;
119  Eigen::Matrix<double, StateDim, StateDim> G0 = B * Rinv * B.transpose();
120  Eigen::Matrix<double, StateDim, StateDim> H0 = C.transpose() * Q * C;
121  Eigen::Matrix<double, StateDim, StateDim> A1, G1, H1;
122  for(int i = 0; i < iterMax; i++)
123  {
124  Eigen::Matrix<double, StateDim, StateDim> I_GH_inv =
125  (Eigen::Matrix<double, StateDim, StateDim>::Identity() + G0 * H0).inverse();
126  A1 = A0 * I_GH_inv * A0;
127  G1 = G0 + A0 * I_GH_inv * G0 * A0.transpose();
128  H1 = H0 + A0.transpose() * H0 * I_GH_inv * A0;
129 
130  double relNorm = (H1 - H0).norm() / H1.norm();
131  if(relNorm < relNormThre)
132  {
133  break;
134  }
135  else if(i == iterMax - 1)
136  {
137  ROS_WARN_STREAM("[PreviewControl] Solution of Riccati equation did not converged: " << relNorm << " > "
138  << relNormThre);
139  }
140 
141  A0 = A1;
142  G0 = G1;
143  H0 = H1;
144  }
145  P_ = H1;
146 
147  riccati_error_ = (P_
148  - (A.transpose() * P_ * A + C.transpose() * Q * C
149  - A.transpose() * P_ * B * (R + B.transpose() * P_ * B).inverse() * B.transpose() * P_ * A))
150  .norm();
151 
152  // 2. Calculate the feedback gain (K and f)
153  Eigen::Matrix<double, InputDim, InputDim> R_BtPB_inv = (R + B.transpose() * P_ * B).inverse();
154  // 2.1 Calculate K
155  K_ = R_BtPB_inv * B.transpose() * P_ * A;
156  // 2.2 Calculate f
157  F_.resize(inputDim, horizon_steps_ * outputDim);
158  Eigen::Matrix<double, StateDim, StateDim> A_BK = A - B * K_;
159  Eigen::Matrix<double, StateDim, StateDim> fSub = Eigen::Matrix<double, StateDim, StateDim>::Identity();
160  for(int i = 0; i < horizon_steps_; i++)
161  {
162  if(i < horizon_steps_ - 1)
163  {
164  F_.middleCols(i * outputDim, outputDim) = R_BtPB_inv * B.transpose() * fSub * C.transpose() * Q;
165  }
166  else
167  {
168  // See https://github.com/euslisp/jskeus/pull/551
169  F_.middleCols(i * outputDim, outputDim) = R_BtPB_inv * B.transpose() * fSub * P_ * C.transpose();
170  }
171  fSub = fSub * A_BK.transpose();
172  }
173  }
