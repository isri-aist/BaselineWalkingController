99  for(int dimIdx = 0; dimIdx < dim_; dimIdx++)
100  {
101  // tridiagonalSystem represents the coefficients (A and b) of linear
102  // system A x = b where A is tridiagonal matrix. In each row, first three
103  // elements are the tridiagonal elements of A and the last element is the
104  // element of b.
105  Eigen::MatrixXd tridiagonalSystem(n, 4);
106  Eigen::VectorXd y(n);
107  Eigen::VectorXd yd(n);
108  Eigen::VectorXd deltaT(n - 1);
109  Eigen::VectorXd deltaY(n - 1);
110  Eigen::VectorXd h(n - 1);
111  Eigen::VectorXd r(n - 1);
112 
113  // Set y, deltaT, deltaY, h, r
114  {
115  auto pointIt = points_.begin();
116  for(size_t k = 0; k < n; k++, pointIt++)
117  {
118  y(k) = pointIt->second(dimIdx);
119  if(k != n - 1)
120  {
121  deltaT(k) = std::next(pointIt)->first - pointIt->first;
122  deltaY(k) = std::next(pointIt)->second(dimIdx) - pointIt->second(dimIdx);
123  h(k) = deltaT(k);
124  r(k) = deltaY(k) / deltaT(k);
125  }
126  }
127  }

133  tridiagonalSystem.row(k + 1) << h(k + 1), 2 * (h(k) + h(k + 1)), h(k), 3 * (r(k) * h(k + 1) + r(k + 1) * h(k));

166  for(size_t i = 1; i < n; i++)

169  tridiagonalSystem(i, 1) -= w * tridiagonalSystem(i - 1, 2);

171  }

173  for(int i = static_cast<int>(n - 2); i >= 0; i--)
174  {
